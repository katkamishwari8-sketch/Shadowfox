"""
sentiment_analysis.py

Core module for loading and running DistilBERT sentiment classifier.
"""

from transformers import pipeline


class SentimentAnalyzer:
    def __init__(self, model_name="distilbert-base-uncased-finetuned-sst-2-english"):
        """
        Initialize the sentiment analysis pipeline.
        """
        self.classifier = pipeline(
            "sentiment-analysis",
            model=model_name
        )

    def predict(self, text: str) -> dict:
        """
        Predict sentiment for a single text input.
        """
        result = self.classifier(text)[0]
        return {
            "text": text,
            "label": result["label"],
            "confidence": float(round(result["score"], 4))
        }

    def batch_predict(self, texts: list) -> list:
        """
        Predict sentiment for a list of texts.
        """
        results = self.classifier(texts)
        output = []

        for text, result in zip(texts, results):
            output.append({
                "text": text,
                "label": result["label"],
                "confidence": float(round(result["score"], 4))
            })

        return output


if __name__ == "__main__":
    analyzer = SentimentAnalyzer()

    sample_texts = [
        "I absolutely love this product!",
        "This is the worst experience ever.",
        "The movie was okay, not great.",
        "Great, my phone just died again. Amazing."
    ]

    predictions = analyzer.batch_predict(sample_texts)

    for prediction in predictions:
        print(prediction)
